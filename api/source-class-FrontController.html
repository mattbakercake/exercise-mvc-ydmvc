  3:  * The FrontController class is the heart of the framework interpreting
  4:  * requests and dispatching appropriate controllers
  5:  * 
  6:  * It determines whether controller/action/parameters have been passed to it 
  7:  * and either sets these values or calls the Request Class to extract the values
  8:  * from the url.  The run method dispatches the controller.
  9:  * 
 10:  * @category Core
 11:  * @version 0.2
 12:  * @since 05-03-2013
 13:  * @author Matt Baker <dev@mikesierra.net> 
 14:  * 
 15:  */
 16: class FrontController {
 17:     
 18:     /**
 19:      * The controller being called
 20:      * @var string
 21:      */
 22:     private $controller;
 23:     /**
 24:      * The action being called
 25:      * @var string
 26:      */
 27:     private $action;
 28:     /**
 29:      * Array of parameters being passed to controller
 30:      * @var array 
 31:      */
 32:     private $params = array();
 33:     
 34:     /**
 35:      * Constructor determines whether controller/action/parameters have been
 36:      * passed to it and either sets controller/action/parameters or
 37:      * parses the url to get this information.
 38:      * 
 39:      * @param Array $options Array of front controller options e.g. 
 40:      * ('controller'=>'index','action'=>'showusers','params'=>array('1','2'))
 41:      * 
 42:      * @example
 43:      *  $FrontController = new FrontController();<br/>
 44:      *      ##OR##<br/>
 45:      *  $FrontController = new FrontController(array(<br/>
 46:      *                                          'controller'=>'index',<br/>
 47:      *                                          'action'=>'showusers',<br/>
 48:      *                                          'params=>array('1','2')<br/>
 49:      *                                         ));<br/>
 50:      */
 51:     function __construct(array $options = array()) {
 52:         
 53:         if (empty($options)) { //if no options passed to object
 54:             $request = new Request();//request object decodes url
 55:         
 56:             //set controller,action and value properties
 57:             if (!is_null($request->controller)) {
 58:                 $this->controller = $request->controller;
 59:              }
 60:             if (isset($request->action)) {
 61:                 $this->action = $request->action;
 62:             }
 63:             if (isset($request->params)) {
 64:                 $this->params = $request->params;
 65:             }
 66:         } else { //if options have been passed to object set them
 67:            if (isset($options['controller'])) {
 68:                $this->controller = $options['controller'];
 69:            }
 70:            if (isset($options['action'])) {
 71:                $this->action = $options['action'];
 72:            }
 73:            if (isset($options['params'])) {
 74:                $this->params = $options['params'];
 75:            }
 76:            
 77:         }
 78:     }
 79:     
 80:     /**
 81:      * Runs the front controller for request.  Creates a new instance of 
 82:      * the controller and action that has been set
 83:      * by the url request
 84:      */
 85:     public function run() {
 86:       if (isset($this->controller)) { //check controller is set
 87:           if (isset($this->action)) { //check action is set
 88:                 $className = ucwords($this->controller) . '_Controller';
 89:                 if (class_exists($className)) { //check class file exists
 90:                     new $className($this->action,$this->params); //instantiate 
 91:                 } else {
 92:                     //no controller class defined
 93:                     throw new Exception(' Controller class not defined: '.$className. ' ');
 94:                 }
 95:           } else {
 96:               //no action defined
 97:              throw new Exception(' Controller action not set ');
 98:           }
 99:       } else {
100:           //no controller set
101:           throw new Exception(' Controller not set ');
102:       }
103:     }
104:     
105: }
